program Slednev4_2;
Uses
  SysUtils;
Const
  m=4;
  z:array[1..m,1..m] of real=((-4.1,3.14,5.21,-1),
                                     (1.1,2.5,1.15,13),
                               (4.5,5.4,-3.7,21),
                         (1.8,0.5,-0.5,-6));


var
  i,j,k,l:integer;
  t:array[1..4] of integer;
  mas_t:array[1..4]of real;
  mas_o:array[1..16] of real;
sum:real;
begin
for i:=1 to 4 do
begin
   for j:=1 to 4 do
   write(z[i,j]:7:2);
writeln;
end;
k:=0;

for i:=1 to 4 do
   for j:=1 to 4 do
      if (i=j) and (z[i,j]<0) then begin
                         k:=k+1;
                         t[k]:=i;
                         end;
writeln('masiv iz nomerov strok gl. diagonali');
for j:=1 to k do
write(t[j]:2);
writeln;
writeln('masiv iz sum strok gl. diagonali');
for i:=1 to k do
begin
  sum:=0;
  for j:=1 to 4 do
   sum:=sum+z[t[i],j];
mas_t[i]:=sum;
end;
for j:=1 to k do
write(mas_t[j]:7:2);
writeln;
writeln('odnomern.masiv strok gl. diagonali');
l:=0;
for i:=1 to k do
    for j:=1 to 4 do
      begin
      l:=l+1;
      mas_o[l]:=z[t[i],j];
      end;
for j:=1 to l do
write(mas_o[j]:7:2);
writeln;
readln;
end.
